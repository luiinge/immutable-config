package imconfig.internal;

import imconfig.*;
import java.util.*;
import java.util.function.*;
import java.util.stream.Stream;

public class EmptyConfig implements Config {

    final static Config INSTANCE = new EmptyConfig();


    @Override
    public Config alteringKeys(UnaryOperator<String> alterOperation) {
        return this;
    }


    @Override
    public Config filtered(Predicate<String> filter) {
        return this;
    }


    @Override

        return this;
    }


    @Override

        return this;
    }

        return true;
    }

        return false;
    }

        return false;
    }

        return List.of();
    }


    @Override

        return Collections.emptyIterator();
    }


    @Override
    public Stream<String> keyStream() {
        return Stream.empty();
    }


    @Override

        return Optional.empty();
    }

        return null;

    @Override
    public Properties asProperties() {
        return new Properties();

    @Override

        return Map.of();
